// utils/buildEntityPayload.js
export function buildEntityPayload(entity) {
  if (!entity) return null;

  // Si ya existe en DB
  if (entity._id) return entity._id;

  // Si es nuevo
  return {
    ...entity,
  };
}

Payload de Orden

export function buildOrdenPayload(formState = {}) {
  const { cliente, equipo, tecnico, orden = {} } = formState;

  const {
    lineasServicio = [],
    total = 0,
    fechaIngreso = new Date().toISOString(),
    diagnosticoCliente = '',
    observaciones = '',
  } = orden;

  return {
    cliente: buildEntityPayload(cliente),
    equipo: buildEntityPayload(equipo),
    tecnico: buildEntityPayload(tecnico),

    lineasServicio: lineasServicio.map((l) => ({
      tipoTrabajo: buildEntityPayload(l?.tipoTrabajo),
      descripcion: l?.descripcion ?? '',
      precioUnitario: Number(l?.precioUnitario ?? 0),
      cantidad: Number(l?.cantidad ?? 1),
    })),

    total: Number(total),
    fechaIngreso,
    diagnosticoCliente,
    observaciones,
  };
}

Helper para resolver refs

async function resolveRef(Model, payload) {
  if (!payload) return null;

  // Ya existe
  if (typeof payload === 'string' || payload._id) {
    return payload;
  }

  // Es nuevo
  const doc = await Model.create(payload);
  return doc._id;
}
